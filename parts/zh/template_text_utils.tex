\newcommand{\enquote}[1]{``{#1}''}

% 字体无关的勾选框，避免在 strict_pku 下出现勾偏到框外的问题
\newcommand{\thesischeckboxsize}{0.88em}
\newcommand{\thesischeckboxrule}{0.06em}
\newcommand{\thesischeckrule}{0.10em}
\newcommand{\thesischeckboxbase}{-0.74ex}
\newcommand{\thesischeckboxnode}[1]{%
	\tikz[baseline=\thesischeckboxbase, x=1em, y=1em]{#1}%
}
% 方框
\renewcommand{\Box}{%
	\thesischeckboxnode{%
		\draw[line width=\thesischeckboxrule]
			(0,0) rectangle (\thesischeckboxsize,\thesischeckboxsize);
	}%
}
% 对勾
\newcommand{\checkbox}{%
	\thesischeckboxnode{%
		\draw[line width=\thesischeckrule, line cap=round, line join=round]
			(0.18,0.42) -- (0.36,0.22) -- (0.72,0.68);
	}%
}
% 勾选了对勾的方框
\newcommand{\checkedbox}{%
	\thesischeckboxnode{%
		\draw[line width=\thesischeckboxrule]
			(0,0) rectangle (\thesischeckboxsize,\thesischeckboxsize);
		\draw[line width=\thesischeckrule, line cap=round, line join=round]
			(0.18,0.42) -- (0.36,0.22) -- (0.72,0.68);
	}%
}

\newsavebox{\thesisdegreetyperefbox}

\newlength{\thesisdegreetypeshift}
\newcommand{\degreetypealignedmark}[1]{%
	\sbox{\thesisdegreetyperefbox}{{\fangsong\zihao{3}学}}%
	\sbox{\thesisdegreetypemarkbox}{{#1}}%
	\setlength{\thesisdegreetypeshift}{%
		\dimexpr(\ht\thesisdegreetyperefbox-\dp\thesisdegreetyperefbox-\ht\thesisdegreetypemarkbox+\dp\thesisdegreetypemarkbox)/2\relax
	}%
	\raisebox{\thesisdegreetypeshift}{\usebox{\thesisdegreetypemarkbox}}%
}

\ExplSyntaxOn
\seq_new:N \l__thesis_keyword_items_seq
\seq_new:N \l__thesis_keyword_clean_seq
\seq_new:N \l__thesis_keyword_output_seq
\tl_new:N \l__thesis_keyword_input_tl
\tl_new:N \l__thesis_keyword_item_tl

\cs_new_protected:Npn \thesis_keywords_collect:n #1
{
	\tl_set:Nx \l__thesis_keyword_input_tl {#1}
	\regex_replace_all:nnN {\s*，\s*} {,} \l__thesis_keyword_input_tl
	\regex_replace_all:nnN {\s*,\s*} {,} \l__thesis_keyword_input_tl
	\tl_trim_spaces:N \l__thesis_keyword_input_tl
	\seq_set_split:NnV \l__thesis_keyword_items_seq {,} \l__thesis_keyword_input_tl
	\seq_clear:N \l__thesis_keyword_clean_seq
	\seq_map_inline:Nn \l__thesis_keyword_items_seq {
		\tl_set:Nn \l__thesis_keyword_item_tl {##1}
		\tl_trim_spaces:N \l__thesis_keyword_item_tl
		\tl_if_blank:VF \l__thesis_keyword_item_tl {
			\seq_put_right:NV \l__thesis_keyword_clean_seq \l__thesis_keyword_item_tl
		}
	}
}

\cs_new_protected:Npn \thesis_format_zh_keywords:n #1
{
	\thesis_keywords_collect:n {#1}
	\seq_use:Nn \l__thesis_keyword_clean_seq {，}
}

\cs_new_protected:Npn \thesis_format_en_keywords:n #1
{
	\thesis_keywords_collect:n {#1}
	\seq_clear:N \l__thesis_keyword_output_seq
	\seq_map_inline:Nn \l__thesis_keyword_clean_seq {
		\tl_set:Nn \l__thesis_keyword_item_tl {##1}
		\tl_set:Nx \l__thesis_keyword_item_tl
			{
				\tl_upper_case:n { \tl_head:V \l__thesis_keyword_item_tl }
				\tl_tail:V \l__thesis_keyword_item_tl
			}
		\seq_put_right:NV \l__thesis_keyword_output_seq \l__thesis_keyword_item_tl
	}
	\seq_use:Nn \l__thesis_keyword_output_seq {,\space}
}
