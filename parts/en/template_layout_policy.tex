	\scalebox{0.68}{\tikz[baseline = {([yshift = -0.08\ccwd] char.base)}]{
		\node[
			shape = circle,
			draw = black,
			minimum size = 1.05\ccwd,
			inner sep = 0pt
		] (char) {#1};
	}}%

		\fontsize{10.5bp}{16bp}\selectfont
		\setlength{\parindent}{0pt}%
		\setlength{\parskip}{0pt}%
		\noindent #1\par
	\endgroup
}

% ============================================================
% 著者-年份制页码上角标
% ============================================================
\@ifpackageloaded{biblatex}{%
	\ifthenelse{\equal{\bibmode}{author_year}}{%
		\let\thesis@orig@parencite\parencite
		\let\thesis@orig@textcite\textcite
		\let\thesis@orig@parencites\parencites
		\let\thesis@orig@textcites\textcites

		\newcommand{\thesis@cite@printsup}[1]{%
			\ifstrempty{#1}{}{%
				\textsuperscript{#1}%
			}%
		}

		\newcommand{\thesis@cite@withlocale}[1]{#1}

		\newcommand{\thesis@cite@single}[4]{%
			\IfNoValueTF{#2}{%
				\thesis@cite@withlocale{#1{#4}}%
			}{%
				\IfNoValueTF{#3}{%
					\thesis@cite@withlocale{#1{#4}}%
					\thesis@cite@printsup{#2}%
				}{%
					\thesis@cite@withlocale{#1[#2][]{#4}}%
					\thesis@cite@printsup{#3}%
				}%
			}% 
		}

		\newcommand{\thesis@cite@twice}[7]{%
			\def\thesis@prenotea{}%
			\def\thesis@postnotea{}%
			\def\thesis@prenoteb{}%
			\def\thesis@postnoteb{}%
			\IfNoValueTF{#2}{}{%
				\IfNoValueTF{#3}{%
					\def\thesis@postnotea{#2}%
				}{%
					\def\thesis@prenotea{#2}%
					\def\thesis@postnotea{#3}%
				}%
			}% 
			\IfNoValueTF{#5}{}{%
				\IfNoValueTF{#6}{%
					\def\thesis@postnoteb{#5}%
				}{%
					\def\thesis@prenoteb{#5}%
					\def\thesis@postnoteb{#6}%
				}%
			}% 
			\thesis@cite@withlocale{%
				\ifstrempty{\thesis@prenotea}{%
					\ifstrempty{\thesis@prenoteb}{%
						#1{#4}{#7}%
					}{%
						#1{#4}[\thesis@prenoteb][]{#7}%
					}% 
				}{%
					\ifstrempty{\thesis@prenoteb}{%
						#1[\thesis@prenotea][]{#4}{#7}%
					}{%
						#1[\thesis@prenotea][]{#4}[\thesis@prenoteb][]{#7}%
					}% 
				}% 
			}
			\begingroup
			\def\thesis@notes{}%
			\ifstrempty{\thesis@postnotea}{}{%
				\edef\thesis@notes{\thesis@postnotea}%
			}% 
			\ifstrempty{\thesis@postnoteb}{}{%
				\ifstrempty{\thesis@notes}{%
					\edef\thesis@notes{\thesis@postnoteb}%
				}{%
					\edef\thesis@notes{\thesis@notes;\thesis@postnoteb}%
				}%
			}% 
			\thesis@cite@printsup{\thesis@notes}%
			\endgroup
		}

		\RenewDocumentCommand{\parencite}{o o m}{%
			\thesis@cite@single{\thesis@orig@parencite}{#1}{#2}{#3}%
		}

		\RenewDocumentCommand{\textcite}{o o m}{%
			\thesis@cite@single{\thesis@orig@textcite}{#1}{#2}{#3}%
		}

		\@ifundefined{citep}{}{%
			\let\thesis@orig@citep\citep
			\RenewDocumentCommand{\citep}{o o m}{%
				\thesis@cite@single{\thesis@orig@citep}{#1}{#2}{#3}%
			}
		}

		\@ifundefined{citet}{}{%
			\let\thesis@orig@citet\citet
			\RenewDocumentCommand{\citet}{o o m}{%
				\thesis@cite@single{\thesis@orig@citet}{#1}{#2}{#3}%
			}
		}

		\RenewDocumentCommand{\parencites}{o o m o o m}{%
			\thesis@cite@twice{\thesis@orig@parencites}{#1}{#2}{#3}{#4}{#5}{#6}%
		}

		\RenewDocumentCommand{\textcites}{o o m o o m}{%
			\thesis@cite@twice{\thesis@orig@textcites}{#1}{#2}{#3}{#4}{#5}{#6}%
		}
	}{}
}{}
